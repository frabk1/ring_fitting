 20def _img_to_array(img):
 21    """
 22    Get a clean 2D numpy array from the image.
 23
 24    **Args**:
 25    * img (ehtim.Image or similar): image object exposing `imarr()` or `imarr`.
 26
 27    **Returns**:
 28    * arr (np.ndarray): 2D float array (H, W) representing image data.
 29
 30    """
 31    arr = img.imarr()
 32    arr = np.squeeze(arr)
 33    if arr.ndim != 2:
 34        raise ValueError(f"Expected 2D image array, got shape {arr.shape}")
 35    return arr

 37def _flux_center(data):
 38    """
 39    Flux-weighted center (x, y) using nonnegative weights.
 40
 41    **Args**:
 42    * data (np.ndarray): 2D image array (H, W), nonnegative preferred.
 43
 44    **Returns**:
 45    * center (tuple[float, float]): (xc, yc) in pixel coordinates.
 46
 47    """
 48    h, w = data.shape
 49    yy, xx = np.mgrid[0:h, 0:w]
 50    wts = np.maximum(data, 0.0)
 51    tot = wts.sum()
 52    if tot <= 0:
 53        return (w - 1) / 2.0, (h - 1) / 2.0
 54    xc = (wts * xx).sum() / tot
 55    yc = (wts * yy).sum() / tot
 56    return float(xc), float(yc)

 58def _estimate_background(data, patch_frac=0.12):
 59    """
 60    Quick background estimate from the darkest corner patches.
 61
 62    **Args**:
 63    * data (np.ndarray): 2D image array (H, W).
 64    * patch_frac (float): fractional patch size based on min(H, W).
 65
 66    **Returns**:
 67    * background (float): estimated background level.
 68
 69    """
 70    h, w = data.shape
 71    k = max(1, int(patch_frac * min(h, w)))
 72    patches = [
 73        data[0:k, 0:k],
 74        data[0:k, w - k:w],
 75        data[h - k:h, 0:k],
 76        data[h - k:h, w - k:w],
 77    ]
 78    means = [p.mean() for p in patches]
 79    return float(min(means))

 81def _radial_profile(data, xc, yc, bin_size=1.0):
 82    """
 83    Azimuthal average vs radius around (xc, yc).
 84
 85    **Args**:
 86    * data (np.ndarray): 2D image array (H, W).
 87    * xc (float): x-coordinate of center.
 88    * yc (float): y-coordinate of center.
 89    * bin_size (float): radial bin size in pixels.
 90
 91    **Returns**:
 92    * r_centers (np.ndarray): radius centers for bins.
 93    * prof (np.ndarray): azimuthally averaged profile at r_centers.
 94
 95    """
 96    h, w = data.shape
 97    yy, xx = np.mgrid[0:h, 0:w]
 98    rr = np.sqrt((xx - xc) ** 2 + (yy - yc) ** 2).ravel()
 99    vv = data.ravel()
100    rmax = rr.max()
101    nbins = max(1, int(np.ceil((rmax + 1e-9) / bin_size)))
102    idx = np.clip((rr / bin_size).astype(int), 0, nbins - 1)
103    sum_v = np.bincount(idx, weights=vv, minlength=nbins)
104    cnt_v = np.bincount(idx, minlength=nbins)
105    with np.errstate(invalid="ignore", divide="ignore"):
106        prof = sum_v / np.maximum(cnt_v, 1)
107    r_centers = (np.arange(nbins) + 0.5) * bin_size
108    return r_centers, prof

110def _smooth_profile(prof):
111    """
112    Tiny 1D smooth to quiet pixel noise.
113
114    **Args**:
115    * prof (np.ndarray): 1D profile array.
116
117    **Returns**:
118    * smoothed (np.ndarray): smoothed profile, same shape as input.
119
120    """
121    k = np.array([1, 2, 3, 2, 1], dtype=float)
122    k /= k.sum()
123    return np.convolve(prof, k, mode="same")

125def _fwhm_width(r, prof, peak_idx, background):
126    """
127    FWHM-style width around a peak; falls back to moment width.
128
129    **Args**:
130    * r (np.ndarray): radius array (same length as prof).
131    * prof (np.ndarray): profile values over r.
132    * peak_idx (int): index of the peak in prof.
133    * background (float): background level for half-max.
134
135    **Returns**:
136    * width (float): estimated FWHM-like width in pixel units.
137
138    """
139    p_peak = prof[peak_idx]
140    half = background + 0.5 * (p_peak - background)
141    i_left = None
142    for i in range(peak_idx, -1, -1):
143        if prof[i] <= half:
144            i_left = i
145            break
146    i_right = None
147    for i in range(peak_idx, len(prof)):
148        if prof[i] <= half:
149            i_right = i
150            break
151    if i_left is not None and i_right is not None and i_right > i_left:
152        def interp_x(i0, i1):
153            y0, y1 = prof[i0], prof[i1]
154            x0, x1 = r[i0], r[i1]
155            if y1 == y0:
156                return x0
157            t = (half - y0) / (y1 - y0)
158            return x0 + t * (x1 - x0)
159        rL = interp_x(i_left - 1 if i_left > 0 else i_left, i_left)
160        rR = interp_x(i_right, i_right + 1 if i_right + 1 < len(prof) else i_right)
161        return float(max(1e-6, rR - rL))
162    wL = max(0, peak_idx - 3)
163    wR = min(len(prof), peak_idx + 4)
164    rw = r[wL:wR]
165    pw = np.maximum(prof[wL:wR] - background, 0.0)
166    sw = pw.sum()
167    if sw <= 0:
168        return 2.0
169    mu = (rw * pw).sum() / sw
170    var = (pw * (rw - mu) ** 2).sum() / sw
171    return float(max(2.0, 2.355 * np.sqrt(max(var, 1e-12))))

173def estimate_ring_parameters(img, bin_size=1.0, patch_frac=0.12, n_theta=180, step=0.5):
174    """
175    Fast ring guesses: radius, width, peak, background, and center.
176
177    **Args**:
178    * img (ehtim.Image or similar): input image object.
179    * bin_size (float): radial bin size for fallback profile.
180    * patch_frac (float): corner patch fraction for background estimate.
181    * n_theta (int): number of rays for radial scans.
182    * step (float): step size along each ray (pixels).
183
184    **Returns**:
185    * radius (float): median peak radius across rays or from fallback profile.
186    * width (float): median FWHM-like width across rays or from fallback.
187    * peak_value (float): maximum peak level found.
188    * background (float): estimated background level.
189    * center (tuple[float, float]): (xc, yc) using image midpoint.
190
191    """
192    data = _img_to_array(img)
193    h, w = data.shape
194    bkg = _estimate_background(data, patch_frac=patch_frac)
195
196    # center = image middle (as requested)
197    xc, yc = (w - 1) / 2.0, (h - 1) / 2.0
198
199    rmax = float(np.hypot(h, w))
200    thetas = np.linspace(0.0, 2.0 * np.pi, int(n_theta), endpoint=False)
201
202    radii = []
203    widths = []
204    peak_vals = []
205
206    k = np.array([1, 2, 3, 2, 1], float)  # light smoothing for 1D ray profiles
207    k /= k.sum()
208
209    for th in thetas:
210        rs = np.arange(0.0, rmax, float(step))
211        xs = xc + rs * np.cos(th)
212        ys = yc + rs * np.sin(th)
213
214        m = (xs >= 0) & (xs <= w - 1) & (ys >= 0) & (ys <= h - 1)
215        if not np.any(m):
216            continue
217        xs, ys, rs = xs[m], ys[m], rs[m]
218        if rs.size < 3:
219            continue
220
221        ix = np.rint(xs).astype(int)
222        iy = np.rint(ys).astype(int)
223        prof = data[iy, ix]
224        prof_s = np.convolve(prof, k, mode="same")
225
226        j = int(np.nanargmax(prof_s))
227        vmax = float(prof_s[j])
228        peak_vals.append(vmax)
229        radii.append(float(rs[j]))
230
231        half = bkg + 0.5 * (vmax - bkg)
232
233        jl = j
234        while jl > 0 and prof_s[jl] > half:
235            jl -= 1
236        jr = j
237        L = len(prof_s)
238        while jr < L - 1 and prof_s[jr] > half:
239            jr += 1
240
241        def _interp_r(i0, i1):
242            y0, y1 = prof_s[i0], prof_s[i1]
243            r0, r1 = rs[i0], rs[i1]
244            if y1 == y0:
245                return r0
246            t = (half - y0) / (y1 - y0)
247            return r0 + t * (r1 - r0)
248
249        if jl > 0 and jr < L - 1:
250            rL = _interp_r(jl, jl + 1)
251            rR = _interp_r(jr - 1, jr)
252            widths.append(float(max(1e-6, rR - rL)))
253
254    if len(radii) == 0:
255        """
256        Fallback: use global radial profile if rays fail.
257        """
258        r, prof = _radial_profile(data, xc, yc, bin_size=bin_size)
259        ps = _smooth_profile(prof)
260        idx = int(np.nanargmax(ps))
261        radius = float(r[idx])
262        width = _fwhm_width(r, ps, idx, bkg)
263        peak_value = float(ps[idx])
264    else:
265        radius = float(np.median(radii))
266        width = float(np.median(widths)) if widths else max(2.0, 0.1 * radius)
267        peak_value = float(np.max(peak_vals))
268
269    return radius, width, peak_value, float(bkg), (xc, yc)

271def rbp_find_bright_points(img, threshold=None, radius=None, margin=None, max_it=999):
272    """
273    Recursive brightest-point finder with circular blanking.
274
275    **Args**:
276    * img (ehtim.Image or similar): input image object.
277    * threshold (float or None): minimum brightness to accept a point; if None, auto.
278    * radius (float or None): blanking radius in pixels; if None, auto from width.
279    * margin (int or None): border margin to ignore; if None, derived from radius.
280    * max_it (int): maximum number of points to extract.
281
282    **Returns**:
283    * points (np.ndarray): shape (N, 2) array of (x, y) pixel coordinates.
284
285    """
286    data = _img_to_array(img)
287    if threshold is None or radius is None:
288        r0, w0, pmax, bkg, _ = estimate_ring_parameters(img)
289        if threshold is None:
290            threshold = bkg + 0.5 * (pmax - bkg)
291        if radius is None:
292            radius = max(1.0, 3.0 * w0)
293    h, w = data.shape
294    if margin is None:
295        margin = int(np.ceil(radius + 1))
296    mask = np.ones_like(data, dtype=bool)
297    points = []
298    for _ in range(max_it):
299        masked = data * mask
300        peak = masked.max()
301        if peak < threshold:
302            break
303        y, x = np.unravel_index(np.argmax(masked), data.shape)
304        if x < margin or x >= (w - margin) or y < margin or y >= (h - margin):
305            mask[y, x] = False
306            continue
307        points.append((x, y))
308        y0, y1 = max(0, y - int(radius)), min(h, y + int(radius) + 1)
309        x0, x1 = max(0, x - int(radius)), min(w, x + int(radius) + 1)
310        yy, xx = np.ogrid[y0:y1, x0:x1]
311        dist = np.sqrt((xx - x) ** 2 + (yy - y) ** 2)
312        mask[y0:y1, x0:x1][dist <= radius] = False
313    return np.array(points)

318def compute_ring_features(angles, radii, brightness, bins=360):
319    """
320    Compute single-number diagnostics from per-angle ring samples.
321
322    **Args**:
323    * angles (array-like): angles in radians for each sample.
324    * radii (array-like): radius at each angle (pixels).
325    * brightness (array-like): brightness/intensity per sample.
326    * bins (int): even number of angular bins for pairing and widths.
327
328    **Returns**:
329    * features (dict): {
330        mean_radius, std_radius,
331        radial_asymmetry, brightness_asymmetry,
332        angular_brightness_R, angular_brightness_circstd_rad,
333        angular_brightness_circstd_deg, angular_brightness_FWHM_deg,
334        samples_used, bins
335      }
336
337    """
338    a = np.asarray(angles, float).ravel()
339    r = np.asarray(radii, float).ravel()
340    b = np.asarray(brightness, float).ravel()
341    m = np.isfinite(a) & np.isfinite(r) & np.isfinite(b)
342    a, r, b = a[m] % TAU, r[m], b[m]
343    if a.size < 8 or (bins % 2): 
344        raise ValueError("Need >=8 samples and even `bins`.")
345    mean_radius, std_radius = float(r.mean()), float(r.std())
346
347    # Uniform angular bins
348    edges = np.linspace(0, TAU, bins+1)
349    idx = np.digitize(a, edges, right=False) - 1
350    idx[idx == bins] = 0
351    cnt = np.bincount(idx, minlength=bins)
352    rsum = np.bincount(idx, weights=r, minlength=bins)
353    bsum = np.bincount(idx, weights=b, minlength=bins)
354    r_u = rsum / np.maximum(1, cnt)
355    b_u = bsum.astype(float)
356    th = 0.5*(edges[:-1] + edges[1:])
357
358    # Opposite-angle asymmetries
359    def _opp_asym(x, norm):
360        d = np.abs(x - np.roll(x, x.size//2)).mean()
361        return float(d / (norm if norm > 0 else 1.0))
362    radial_asym = _opp_asym(r_u, mean_radius)
363    mean_b = float(b_u.mean())
364    brightness_asym = _opp_asym(b_u, mean_b)
365
366    # Circular concentration & circular std (brightness-weighted)
367    W = b_u.sum()
368    if W <= 0 or not np.isfinite(W):
369        R = 0.0; circstd_rad = np.nan
370    else:
371        C = float((b_u*np.cos(th)).sum()/W)
372        S = float((b_u*np.sin(th)).sum()/W)
373        R = float(np.hypot(C, S))
374        circstd_rad = (np.sqrt(max(0.0, -2*np.log(R))) if R > 0 else np.inf)
375    circstd_deg = float(np.degrees(circstd_rad)) if np.isfinite(circstd_rad) else np.nan
376
377    # FWHM (deg) of largest brightness peak (simple half-max crossing)
378    def _fwhm_deg(y):
379        n = y.size
380        k = int(np.argmax(y)); ymax = float(y[k])
381        if not np.isfinite(ymax) or ymax <= 0: return np.nan
382        half = 0.5*ymax
383        def cross(start, step):
384            i = start
385            for _ in range(n):
386                j = (i + step) % n
387                yi, yj = y[i], y[j]
388                if (yi - half)*(yj - half) <= 0 and yi != yj:
389                    frac = (half - yi)/(yj - yi)
390                    return (i + frac*step) % n
391                i = j
392            return None
393        L, Rr = cross(k, -1), cross(k, +1)
394        if L is None or Rr is None: return np.nan
395        dth = TAU/n
396        w = ((Rr - L) % n)*dth
397        if w > np.pi: w = TAU - w
398        return float(np.degrees(w))
399    fwhm = _fwhm_deg(b_u)
400
401    return {
402        "mean_radius": mean_radius,
403        "std_radius": std_radius,
404        "radial_asymmetry": radial_asym,
405        "brightness_asymmetry": brightness_asym,
406        "angular_brightness_R": R,
407        "angular_brightness_circstd_rad": float(circstd_rad),
408        "angular_brightness_circstd_deg": float(circstd_deg),
409        "angular_brightness_FWHM_deg": float(fwhm),
410        "samples_used": int(a.size),
411        "bins": int(bins),
412    }
